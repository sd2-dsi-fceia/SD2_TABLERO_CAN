136static void canmsg_escritura(void) {
137 ERROR_t estado;
138
139 // Estado de los pines
140 byte1 = !GPIO_PinRead(SW1_GPIO, SW1_PIN);
141 byte2 = !GPIO_PinRead(SW3_GPIO, SW3_PIN);
142
143 // Mensaje
144 canMsg1.data[0] = byte1;
145 canMsg1.data[1] = byte2;
146
147 estado = mcp2515_sendMessage(&canMsg1);
148
149 if (estado == ERROR_OK) {
150 PRINTF("\nMensaje enviado\n\r");
151 PRINTF("ID\tDLC\tDATA\n\r");
152 PRINTF("%d\t%d\t", canMsg1.can_id, canMsg1.can_dlc);
153
154 for (uint8_t i = 0; i < 8; i++) {
155 PRINTF("%d ", canMsg1.data[i]);
156 }
157 PRINTF("\n\r");
158 } else {
159 PRINTF("\nError al enviar\n\r");
160 }
161
162 return;
163}

165static void perifericos_init(void) {
166 ERROR_t error;
167
168 error = mcp2515_reset(); // Configura el modulo
169
170 if (error != ERROR_OK)
171 PRINTF("Fallo al resetear el modulo\n\r");
172
173 error = mcp2515_setBitrate(CAN_125KBPS, MCP_8MHZ);
174
175 if (error != ERROR_OK)
176 PRINTF("Fallo al setear el bit rate\n\r");
177
178 error = mcp2515_setNormalMode();
179 if (error != ERROR_OK)
180 PRINTF("Fallo al setear el modo normal\n\r");
181
182 // Configura los botones
183 CLOCK_EnableClock(kCLOCK_PortC); // Habilitar reloj para el puerto C
184
185 port_pin_config_t config_switch = {
186 .pullSelect = kPORT_PullUp, // Activar Pull-Up interno
187 .mux = kPORT_MuxAsGpio // Configurar como GPIO
188 };
189
190 PORT_SetPinConfig(PORTC, SW1_PIN, &config_switch);
191 PORT_SetPinConfig(PORTC, SW3_PIN, &config_switch);
192
193 gpio_pin_config_t gpio_config = {
194 .pinDirection = kGPIO_DigitalInput,
195 .outputLogic = 0
196 };
197
198 GPIO_PinInit(SW1_GPIO, SW1_PIN, &gpio_config);
199 GPIO_PinInit(SW3_GPIO, SW3_PIN, &gpio_config);
200
201 return;
202}
