733extern void mcp2515_init(void)
734{
735 /*
736 * Inicializacion de pines.
737 *
738 * Configura los pines de entrada y salida para el modulo spi
739 * de la kl46z. Tengamos en cuenta que los pines se pueden configurar
740 * desde el ConfigTools, al igual que la frecuencia del reloj.
741 *
742 * MISO:PTE19
743 * MOSI:PTE18
744 * SCK: PTE17
745 * CS: PTE16
746 * */
747
748 /* Port E Clock Gate Control: Clock enabled */
749 CLOCK_EnableClock(kCLOCK_PortE);
750
751 gpio_pin_config_t Cs_mcp2515_config = {.pinDirection = kGPIO_DigitalOutput,
752 .outputLogic = 1U};
753 /* Initialize GPIO functionality on pin PTE16 (pin 14) */
754 GPIO_PinInit(BOARD_Cs_mcp2515_GPIO, BOARD_Cs_mcp2515_PIN,
755 &Cs_mcp2515_config);
756
757 /* PORTE16 (pin 14) is configured as PTE16 */
758 PORT_SetPinMux(BOARD_Cs_mcp2515_PORT, BOARD_Cs_mcp2515_PIN,
759 kPORT_MuxAsGpio);
760
761 /* PORTE17 (pin 15) is configured as SPI0_SCK */
762 PORT_SetPinMux(PORTE, 17U, kPORT_MuxAlt2);
763
764 /* PORTE18 (pin 16) is configured as SPI0_MOSI */
765 PORT_SetPinMux(PORTE, 18U, kPORT_MuxAlt2);
766
767 /* PORTE19 (pin 17) is configured as SPI0_MISO */
768 PORT_SetPinMux(PORTE, 19U, kPORT_MuxAlt2);
769
770 CS_HIGH; /* Deselecciona el modulo mcp2515.*/
771
772 /*
773 * Configuracion del spi.
774 *
775 * Se configura el spi para trabajar con/sin freertos. Ver las configuraciones
776 * iniciales en la librería spi.h y spi.c. Acordarse de seter o limpiar USE_FREERTOS.
777 * */
778 spi_init();
779
780 return;
781}

808extern ERROR_t mcp2515_reset(void)
809{
810 mcp2515_init(); // Configura los pines del spi
811
812 ERROR_t error;
813 INSTRUCTION_t inst = INSTRUCTION_RESET;
814
815 /* Reseteamos el modulo */
816 startSPI();
817 status_t status = spi_write(&inst, 1);
818 if (status != kStatus_Success)
819 return ERROR_SPI_WRITE;
820 endSPI();
821
822 __delay_ms(100);
823
824 /* Limpiamos registros de tx y rx*/
825 setRegisters_t setRegs;
826
827#define CANT_REGISTROS 14
828 setRegs.n = CANT_MAX_SET_REGISTERS;
829
830 memset(setRegs.values, 0, CANT_REGISTROS);
831
832 setRegs.reg = MCP_TXB0CTRL;
833 error = mcp2515_setRegisters(setRegs);
834 if (error != ERROR_OK)
835 return error;
836
837 setRegs.reg = MCP_TXB1CTRL;
838
839 error = mcp2515_setRegisters(setRegs);
840 if (error != ERROR_OK)
841 return error;
842 setRegs.reg = MCP_TXB2CTRL;
843
844 error = mcp2515_setRegisters(setRegs);
845 if (error != ERROR_OK)
846 return error;
847
848 setRegister_t setReg;
849
850 setReg.reg = MCP_RXB0CTRL, setReg.value = 0;
851 error = mcp2515_setRegister(setReg);
852 /*
853 * No se detecta el error dado que queremos que
854 * sea cero el registro.
855 * */
856
857 setReg.reg = MCP_RXB1CTRL, setReg.value = 0;
858 error = mcp2515_setRegister(setReg);
859
860 /* Habilitamos interrupciones */
861 setReg.reg = MCP_CANINTE;
862 setReg.value = CANINTF_RX0IF | CANINTF_RX1IF | CANINTF_ERRIF | CANINTF_MERRF;
863 error = mcp2515_setRegister(setReg);
864 if (error != ERROR_OK)
865 return error;
866
867 /* Verificamos que se cargo correctamente */
868 ReadReg_t readReg;
869
870 readReg.reg = MCP_CANINTE;
871 readReg.data = 0;
872
873 error = mcp2515_readRegister(&readReg);
874 if (error != ERROR_OK)
875 return error;
876 if (setReg.value != readReg.data)
877 return ERROR_VERIFICACION_SET_REGISTER;
878
879 /* Configuramos parametros del rx */
880 ModifyReg_t modifyReg;
881
882 modifyReg.reg = MCP_RXB0CTRL;
883 modifyReg.mask = RXBnCTRL_RXM_MASK | RXB0CTRL_BUKT | RXB0CTRL_FILHIT_MASK;
884 modifyReg.data = RXBnCTRL_RXM_STDEXT | RXB0CTRL_BUKT | RXB0CTRL_FILHIT;
885 error = mcp2515_modifyRegister(modifyReg);
886 if (error != ERROR_OK)
887 return error;
888
889 /* Verificamos */
890 readReg.reg = MCP_RXB0CTRL;
891 error = mcp2515_readRegister(&readReg);
892 if ((readReg.data & modifyReg.data) == 0)
893 return ERROR_VERIFICAION_MODIFY_REG;
894
895 modifyReg.reg = MCP_RXB1CTRL;
896 modifyReg.mask = RXBnCTRL_RXM_MASK | RXB1CTRL_FILHIT_MASK;
897 modifyReg.data = RXBnCTRL_RXM_STDEXT | RXB1CTRL_FILHIT;
898 error = mcp2515_modifyRegister(modifyReg);
899 if (error != ERROR_OK)
900 return error;
901
902 /* Limpiamos los filtros y mascaras */
903 RXF filters[] = {RXF0, RXF1, RXF2, RXF3, RXF4, RXF5};
904 for (int i = 0; i < 6; i++)
905 {
906 bool ext = (i == 1);
907 error = mcp2515_setFilter(filters[i], ext, 0);
908 if (error != ERROR_OK)
909 return error;
910 }
911
912 MASK masks[] = {MASK0, MASK1};
913 for (int i = 0; i < 2; i++)
914 {
915 error = mcp2515_setFilterMask(masks[i], true, 0);
916 if (error != ERROR_OK)
917 return error;
918 }
919
920 return ERROR_OK;
921}

1749 * byte 2: EID8
1750 * byte 3: EID0
1751 * byte 4: TXBnDLC
1752 * byte 5-13: Data frame
1753 * */

1766 memcpy(&setRegs.values[MCP_DATA], frame->data, frame->can_dlc);

1769 setRegs.n = 5 + frame->can_dlc;
1770// setRegs.reg = txbuf->SIDH;
1771 setRegs.reg = RegistroTx.TxBSIDH;

1779// modifyReg.reg = txbuf->CTRL;

1785 return error;

1789// .reg = txbuf->CTRL,
1790 .reg = RegistroTx.TxBCTRL,
1791 .data = 0,
1792 };
1793
1794//#if (!USE_FREERTOS)
1795
1796 error = mcp2515_readRegister(&readReg);
1797 if (error != ERROR_OK)
1798 return error;
1799
1800 if ((readReg.data & (TXB_ABTF /*< Message aborted flag bit.*/
1801 | TXB_MLOA /*< Message lost arbitration bit.*/

1844extern ERROR_t mcp2515_readMessageWithBufferId(const RXBn rxbn,
1845 struct can_frame *frame)
1846{
1847 ERROR_t error;
1848 const struct RXBn_REGS *rxb = &RXB[rxbn];
1849
1850 ReadRegs_t readRegs = {
1851 .reg = rxb->SIDH,
1852 .n = 5,
1853 };
1854
1855 /* Lee las id de formato estandar y extendido */
1856 /*
1857 * Se carga el valor de la id estandar y extendida
1858 * al tipo de dato. Para esto debemos cargar el
1859 * registro donde vamos a leer. Este registro
1860 * puede ser al buffer 0 o al 1.
1861 * */
1862 error = mcp2515_readRegisters(&readRegs);
1863 if (error != ERROR_OK)
1864 return error;
1865
1866 /* Tomamos el valor del id */
1867 /*
1868 * Leemos desde el registro sidh. Por tanto tenemos
1869 * SIDH, SIDL, EXID0, EXID8, ...
1870 *
1871 * id = SIDH(8 bits) + SIDL(3 bits estandart + 2 bits extendido) + EXID8(8 bits ext) + EXID0()
1872 * */
1873 uint32_t id = (readRegs.values[MCP_SIDH] << 3) + (readRegs.values[MCP_SIDL] >> 5);
1874
1875 /* Verifica si se enuentra en formato extendido */
1876 if ((readRegs.values[MCP_SIDL] & TXB_EXIDE_MASK) == TXB_EXIDE_MASK)
1877 {
1878 id = (id << 2) + (readRegs.values[MCP_SIDL] & 0x03); /*< Agarra los bits 16 y 17 del exid.*/
1879 id = (id << 8) + readRegs.values[MCP_EID8]; /*< 8 bits de la parte alta.*/
1880 id = (id << 8) + readRegs.values[MCP_EID0]; /*< 8 bits de la parte baja.*/
1881 id |= CAN_EFF_FLAG;
1882 }
1883
1884 /* Determina la cantidad de data frame a recibir */
1885 uint8_t dlc = (readRegs.values[MCP_DLC] & DLC_MASK);
1886
1887 if (dlc > CAN_MAX_DLEN)
1888 return ERROR_FAIL;
1889
1890 /* Leemos el registro de control */
1891 ReadReg_t readReg;
1892
1893 readReg.reg = rxb->CTRL;
1894 readReg.data = 0;
1895 error = mcp2515_readRegister(&readReg);
1896 if (error != ERROR_OK)
1897 return error;
1898
1899 if (readReg.data & RXBnCTRL_RTR)
1900 {
1901 id |= CAN_RTR_FLAG;
1902 }
1903
1904 frame->can_id = id;
1905 frame->can_dlc = dlc;
1906
1907 readRegs.reg = rxb->DATA;
1908 readRegs.n = dlc;
1909 error = mcp2515_readRegisters(&readRegs);
1910 if (error != ERROR_OK)
1911 return error;
1912
1913 memcpy(frame->data, readRegs.values, dlc);
1914
1915 /* Modificamos el registro de interrupciones can */
1916 ModifyReg_t modifyReg = {
1917 .reg = MCP_CANINTF,
1918 .mask = rxb->CANINTF_RXnIF,
1919 .data = 0,
1920 };
1921
1922 error = mcp2515_modifyRegister(modifyReg);
1923 if (error != ERROR_OK)
1924 return error;
1925
1926 return ERROR_OK;
1927}
